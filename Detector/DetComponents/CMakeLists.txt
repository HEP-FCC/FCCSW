                         RecoGeometry
                         FWCore
                         Detector/DetInterface
                         Detector/DetSegmentation
                         Detector/DetCommon
                         Detector/DetExtensions)

find_package(ROOT COMPONENTS MathCore GenVector Geom)

find_package(Geant4)

find_package(ACTS COMPONENTS Core TGeoPlugin DD4hepPlugin)
#workaround for the lone -lGeom flag exported by ACTS

                 src/*.cpp
                 INCLUDE_DIRS GaudiKernel ROOT DD4hep Geant4 RecoGeometry FWCore DetInterface DetExtensions DetSegmentation DetCommon ACTS
                 LINK_LIBRARIES GaudiKernel ROOT DD4hep ${DD4hep_COMPONENT_LIBRARIES} DetSegmentation DetCommon Geant4 RecoGeometry FWCore ACTS::ACTSCore ACTS::ACTSDD4hepPlugin)
